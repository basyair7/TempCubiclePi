# cara merge 2 file sqlite.db
1. buka file dengan DB Browser for SQLite
2. buka tab execute
3. paste kan kode yang dibawah ini

attach 'c:\test\b.db3' as toMerge;           
BEGIN; 
insert into nama_table select * from toMerge.nama_tabel; 
COMMIT; 
detach toMerge;

"nama_tabel" merupakan contoh seperti nama kubikel/nama gardu
